A. Final App Code (ContentView.swift)

Complete ContentView.swift

The full code for the "Magic Me" app after all steps and challenges.

import SwiftUI

struct ContentView: View {
    // The data: An array of quotes
    let quotes = [
        "It's giving 'main character energy'.",
        "Vibe check: passed.",
        "Manifesting this W.",
        "No thoughts, head empty.",
        "Let them cook."
    ]

    // The memory: A state variable to hold the current quote
    @State private var currentQuote = "Tap my face for wisdom..."

    var body: some View {
        ZStack {
            // Background Layer
            LinearGradient(colors: [.purple, .blue], startPoint: .topLeading, endPoint: .bottomTrailing)
                .ignoresSafeArea()

            // Content Layer
            VStack(spacing: 20) {
                // Profile Image with Tap Gesture
                Image("avatar1")
                    .resizable()
                    .aspectRatio(contentMode: .fill)
                    .frame(width: 150, height: 150)
                    .clipShape(Circle())
                    .overlay(Circle().stroke(Color.white, lineWidth: 5))
                    .shadow(radius: 10)
                    .onTapGesture {
                        // Action: Change the quote
                        currentQuote = quotes.randomElement() ?? "Error 404: Wisdom not found."
                    }

                // Name and Title
                VStack {
                    Text("Your Name")
                        .font(.system(size: 32, weight: .bold))
                        .foregroundColor(.white)
                    Text("Future Tech Innovator")
                        .font(.headline)
                        .foregroundColor(.white.opacity(0.8))
                }

                // Quote Display Text
                Text(currentQuote)
                    .font(.title3)
                    .fontWeight(.medium)
                    .multilineTextAlignment(.center)
                    .foregroundColor(.white)
                    .frame(minHeight: 80)
                    .padding()
                    .background(.black.opacity(0.5))
                    .cornerRadius(15)
                    .padding(.horizontal)
                    .animation(.easeInOut(duration: 0.5), value: currentQuote)

                // Hobbies List
                List {
                    HStack {
                        Image(systemName: "gamecontroller.fill")
                        Text("Gaming")
                    }
                    HStack {
                        Image(systemName: "music.note")
                        Text("Listening to Music")
                    }
                    HStack {
                        Image(systemName: "film.fill")
                        Text("Watching Movies")
                    }
                }
                .frame(height: 180)
                .cornerRadius(15)
                .listStyle(.plain)
                .foregroundColor(.blue)
                
                Spacer()
            }
            .padding()
        }
    }
}
